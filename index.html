Мінімальна мова для послідовних обчислень в декартово-замкнених категоріях

Анотація

Намдак ТонпаДАТА: 10 ЖОВТНЯ 2016

Мова програмування — це чисте нетипізоване лямбда-числення, базова модель обчислень, заснована на абстракціях і застосуванні функцій, що використовується як основа переважної більшості мов програмування.

Синтаксис

Терми складаються зі змінних, лямбда-абстракцій і застосувань. Це мінімальна мова, що підтримує обчислення через бета-редукцію.

I = #identifier O = I | ( O ) | O O | λ I -> O
type term = | Var of string | Lam of string * term | App of term * term

Правила обчислень

Єдиним правилом обчислень у є бета-редукція, що замінює застосування лямбда-абстракції на підстановку аргументу.

App (Lam (x, b), a) → subst x a b

Синтаксис

Терми кодуються як алгебраїчний тип даних у мові OCaml.

type term = | Var of string | Lam of string * term | App of term * term

Підстановка

let rec subst x s = function | Var y -> if x = y then s else Var y | Lam (y, t) when x <> y -> Lam (y, subst x s t) | App (f, a) -> App (subst x s f, subst x s a) | t -> t

Рівність

let rec equal t1 t2 = match t1, t2 with | Var x, Var y -> x = y | Lam (x, b), Lam (y, b') -> equal b (subst y (Var x) b') | Lam (x, b), t -> equal b (App (t, Var x)) | t, Lam (x, b) -> equal (App (t, Var x)) b | App (f1, a1), App (f2, a2) -> equal f1 f2 && equal a1 a2 | _ -> false

Редукція

let rec reduce = function | App (Lam (x, b), a) -> subst x a b | App (f, a) -> App (reduce f, reduce a) | t -> t

Нормалізація

let rec normalize t = let t' = reduce t in if equal t t' then t else normalize t'

Внутрішня мова ОЗК

Представлена мова є внутрішньою мовою однооб'єктної замкненої категорії (ОЗК), розширивши яку структурою добутку можна отримати декартово-замкнену категорію (ДЗК).
